–í—Å–µ –¥–∞–Ω–Ω—ã–µ –∞–≥—Ä–µ–≥–∏—Ä–æ–≤–∞–Ω—ã, –∏ –∫–∞–∂–¥–∞—è –Ω–æ–≤–æ—Å—Ç—å –∏–º–µ–µ—Ç –æ—Ü–µ–Ω–∫—É –¥–µ–π—Å—Ç–≤–∏—è –≤ –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏ –æ—Ç —Ç–æ–≥–æ, —á—Ç–æ —Å –Ω–µ–π —Å–¥–µ–ª–∞–ª –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å. –ü—Ä–µ–¥–ø–æ–ª–æ–∂–∏–º, —á—Ç–æ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å –ø—Ä–æ—Å–º–æ—Ç—Ä–µ–ª –∏ –æ—Å—Ç–∞–≤–∏–ª —Å–º–∞–π–ª–∏–∫. –û–±—â–∏–π –±–∞–ª–ª —Å–æ—Å—Ç–∞–≤–∏—Ç: $ -1 $ (–ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—é –±—ã–ª –ø–æ–∫–∞–∑–∞–Ω —ç–ª–µ–º–µ–Ω—Ç) $ + 2 $ (–ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å –ø—Ä–æ—Å–º–æ—Ç—Ä–µ–ª —ç–ª–µ–º–µ–Ω—Ç) $ + 1 $ (–ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å –æ—Å—Ç–∞–≤–∏–ª —Å–º–∞–π–ª–∏–∫) $ = 2 $.

–ü–æ –º–µ—Ä–µ –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤–∏—è –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è —Å —Å–∏—Å—Ç–µ–º–æ–π –º—ã —Ñ–æ—Ä–º–∏—Ä—É–µ–º –µ–≥–æ –≤–µ–∫—Ç–æ—Ä –ø—Ä–µ–¥–ø–æ—á—Ç–µ–Ω–∏–π —Å–ª–µ–¥—É—é—â–∏–º –æ–±—Ä–∞–∑–æ–º:

\begin{equation}
    u_i = \sum_{j=1}^{n} v_j \times s_j
\end{equation}

\begin{itemize}
    \item $u_i$ -- –≤–µ–∫—Ç–æ—Ä $user_i$
    \item $v_j$ -- –≤–µ–∫—Ç–æ—Ä $item_j$
    \item $s_j$ -- –≤–µ—Å –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤–∏—è —Å $item_j$
\end{itemize}

–ü–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å—Å–∫–∏–π –≤–µ–∫—Ç–æ—Ä –∏–º–µ–µ—Ç —Ç–∞–∫—É—é –∂–µ —Ä–∞–∑–º–µ—Ä–Ω–æ—Å—Ç—å, —á—Ç–æ –∏ –≤–µ–∫—Ç–æ—Ä —ç–ª–µ–º–µ–Ω—Ç–∞.

\begin{equation}
    r_{ui}=\cos (\theta)=\frac{\mathbf{u} \cdot \mathbf{v}}{\|\mathbf{u}\|\|\mathbf{v}\|}=\frac{\sum_{i=1}^{n} u_{i} v_{i}}{\sqrt{\sum_{i=1}^{n} u_{i}^{2}} \sqrt{\sum_{i=1}^{n} v_{i}^{2}}}
\end{equation}

\begin{itemize}
    \item $r_{ui}$ -- —Ä–µ–π—Ç–∏–Ω–≥ $user$ $item$
    \item $u$ -- –≤–µ–∫—Ç–æ—Ä $user$
    \item $v$ -- –≤–µ–∫—Ç–æ—Ä $item$
\end{itemize}
